## Introducción

Una API REST (Representational State Transfer) expone recursos del backend a través de endpoints HTTP. Es el estándar de comunicación entre frontend y backend en aplicaciones web modernas.

## Principios REST

| Principio | Descripción | Ejemplo |
|-----------|-------------|---------|
| Recursos | Cada entidad es un recurso | `/usuarios`, `/productos` |
| Verbos HTTP | GET, POST, PUT, PATCH, DELETE | `GET /usuarios` |
| Stateless | Cada petición es independiente | Token JWT en cada request |
| URI única | Cada recurso tiene una URI | `/usuarios/42` |
| Representaciones | JSON, XML, etc. | `Content-Type: application/json` |

## Diseño de endpoints

```bash
# Convención de nombres (plural, sustantivos, kebab-case)
GET    /api/usuarios              # Listar
GET    /api/usuarios/:id          # Obtener uno
POST   /api/usuarios              # Crear
PUT    /api/usuarios/:id          # Actualizar completo
PATCH  /api/usuarios/:id          # Actualizar parcial
DELETE /api/usuarios/:id          # Eliminar

# Anidados para relaciones
GET    /api/usuarios/:id/pedidos  # Pedidos de un usuario
POST   /api/usuarios/:id/pedidos  # Crear pedido para un usuario
```

## API REST con Express

```javascript
const express = require('express');
const app = express();
app.use(express.json());

// Middleware global: logging
app.use((req, res, next) => {
    console.log(`${new Date().toISOString()} ${req.method} ${req.url}`);
    next();
});

// CRUD de usuarios
const usuarios = [];  // en producción: base de datos

// GET /api/usuarios
app.get('/api/usuarios', (req, res) => {
    res.json(usuarios);
});

// GET /api/usuarios/:id
app.get('/api/usuarios/:id', (req, res) => {
    const usuario = usuarios.find(u => u.id === parseInt(req.params.id));
    if (!usuario) return res.status(404).json({ error: 'Usuario no encontrado' });
    res.json(usuario);
});

// POST /api/usuarios
app.post('/api/usuarios', (req, res) => {
    const { nombre, email } = req.body;
    if (!nombre || !email) {
        return res.status(400).json({ error: 'Nombre y email son obligatorios' });
    }
    const nuevo = { id: usuarios.length + 1, nombre, email };
    usuarios.push(nuevo);
    res.status(201).json(nuevo);
});

// PUT /api/usuarios/:id
app.put('/api/usuarios/:id', (req, res) => {
    const idx = usuarios.findIndex(u => u.id === parseInt(req.params.id));
    if (idx === -1) return res.status(404).json({ error: 'No encontrado' });
    usuarios[idx] = { ...usuarios[idx], ...req.body };
    res.json(usuarios[idx]);
});

// DELETE /api/usuarios/:id
app.delete('/api/usuarios/:id', (req, res) => {
    const idx = usuarios.findIndex(u => u.id === parseInt(req.params.id));
    if (idx === -1) return res.status(404).json({ error: 'No encontrado' });
    usuarios.splice(idx, 1);
    res.status(204).send();
});

app.listen(3000);
```

## Autenticación con JWT

JWT (JSON Web Token) es un token firmado que contiene información del usuario.

```bash
npm install jsonwebtoken bcrypt
```

```javascript
const jwt = require('jsonwebtoken');
const bcrypt = require('bcrypt');

const SECRET = process.env.JWT_SECRET || 'clave-secreta-2026';

// Registro
app.post('/api/registro', async (req, res) => {
    const { email, password } = req.body;

    // Hash de contraseña
    const hash = await bcrypt.hash(password, 10);

    // Guardar en BD (simulado)
    const usuario = { id: 1, email, password: hash };

    res.status(201).json({ message: 'Usuario creado' });
});

// Login
app.post('/api/login', async (req, res) => {
    const { email, password } = req.body;

    // Buscar usuario (simulado)
    const usuario = { id: 1, email: 'user@test.com', password: '$2b$10$...' };

    const valido = await bcrypt.compare(password, usuario.password);
    if (!valido) {
        return res.status(401).json({ error: 'Credenciales inválidas' });
    }

    // Generar token
    const token = jwt.sign(
        { id: usuario.id, email: usuario.email },
        SECRET,
        { expiresIn: '24h' }
    );

    res.json({ token, usuario: { id: usuario.id, email: usuario.email } });
});
```

### Middleware de autenticación JWT

```javascript
function autenticar(req, res, next) {
    const header = req.headers.authorization;

    if (!header || !header.startsWith('Bearer ')) {
        return res.status(401).json({ error: 'Token requerido' });
    }

    const token = header.split(' ')[1];

    try {
        const decoded = jwt.verify(token, SECRET);
        req.usuario = decoded;  // disponible en las rutas
        next();
    } catch (error) {
        return res.status(403).json({ error: 'Token inválido o expirado' });
    }
}

// Rutas protegidas
app.get('/api/perfil', autenticar, (req, res) => {
    res.json({ usuario: req.usuario });
});

app.get('/api/usuarios', autenticar, (req, res) => {
    // solo usuarios autenticados pueden ver la lista
    res.json(usuarios);
});
```

## Middleware de validación con express-validator


```javascript
const { body, validationResult } = require('express-validator');

// Validaciones en la ruta
app.post('/api/usuarios',
    body('nombre').trim().isLength({ min: 3 }).withMessage('Mínimo 3 caracteres'),
    body('email').isEmail().withMessage('Email inválido'),
    body('password').isLength({ min: 8 }).withMessage('Mínimo 8 caracteres'),
    (req, res) => {
        const errors = validationResult(req);
        if (!errors.isEmpty()) {
            return res.status(400).json({ errors: errors.array() });
        }
        // Si pasa validación, continuar
        const { nombre, email } = req.body;
        res.status(201).json({ nombre, email });
    }
);
```

## API REST con Spring Boot (Java)

```java
@RestController
@RequestMapping("/api/usuarios")
public class UsuarioController {

    @Autowired
    private UsuarioService service;

    @GetMapping
    public ResponseEntity<List<UsuarioResponseDTO>> listar() {
        return ResponseEntity.ok(service.listar());
    }

    @GetMapping("/{id}")
    public ResponseEntity<UsuarioResponseDTO> obtener(@PathVariable Long id) {
        return ResponseEntity.ok(service.obtener(id));
    }

    @PostMapping
    public ResponseEntity<UsuarioResponseDTO> crear(
            @Valid @RequestBody CrearUsuarioDTO dto) {
        return ResponseEntity.status(201).body(service.crear(dto));
    }

    @DeleteMapping("/{id}")
    public ResponseEntity<Void> eliminar(@PathVariable Long id) {
        service.eliminar(id);
        return ResponseEntity.noContent().build();
    }
}
```

## Códigos de estado HTTP

| Código | Significado | Uso |
|--------|-------------|-----|
| 200 OK | Éxito | GET, PUT, PATCH |
| 201 Created | Recurso creado | POST |
| 204 No Content | Éxito sin cuerpo | DELETE |
| 400 Bad Request | Error del cliente | Validación fallida |
| 401 Unauthorized | No autenticado | Token faltante |
| 403 Forbidden | Sin permisos | Rol insuficiente |
| 404 Not Found | No existe | ID inválido |
| 500 Server Error | Error interno | Excepción no controlada |

## Buenas prácticas

1. **Versionar la API**: `/api/v1/usuarios` para permitir cambios sin romper clientes
2. **Usar códigos HTTP correctos**: no devuelvas 200 para errores
3. **Paginación**: `GET /api/usuarios?page=1&limit=20` con metadatos
4. **Filtros y ordenación**: `?sort=-fecha&estado=activo`
5. **Respuestas consistentes**: formato uniforme para éxito y error
6. **Documentación**: Swagger/OpenAPI para que otros consuman tu API

## Ejercicio práctico

Crea una API REST completa para un gestor de tareas con:
- CRUD de tareas (GET, POST, PUT, DELETE)
- Autenticación JWT (login, registro)
- Middleware de validación
- Documentación Swagger

<summary>Solución Express completa</summary>

```javascript
const express = require('express');
const jwt = require('jsonwebtoken');
const { body, validationResult } = require('express-validator');
require('dotenv').config();

const app = express();
app.use(express.json());

const SECRET = process.env.JWT_SECRET;
let tareas = [];
let usuarios = [];

// Middleware JWT
function auth(req, res, next) {
    const token = req.headers.authorization?.split(' ')[1];
    if (!token) return res.status(401).json({ error: 'Token requerido' });
    try {
        req.user = jwt.verify(token, SECRET);
        next();
    } catch {
        res.status(403).json({ error: 'Token inválido' });
    }
}

// Auth routes
app.post('/api/registro', (req, res) => {
    const { email, password } = req.body;
    usuarios.push({ id: usuarios.length + 1, email, password });
    res.status(201).json({ message: 'Usuario creado' });
});

app.post('/api/login', (req, res) => {
    const { email, password } = req.body;
    const user = usuarios.find(u => u.email === email);
    if (!user || user.password !== password) {
        return res.status(401).json({ error: 'Credenciales inválidas' });
    }
    const token = jwt.sign({ id: user.id, email }, SECRET, { expiresIn: '24h' });
    res.json({ token });
});

// CRUD tareas (protegido)
app.get('/api/tareas', auth, (req, res) => {
    const misTareas = tareas.filter(t => t.userId === req.user.id);
    res.json(misTareas);
});

app.post('/api/tareas', auth,
    body('titulo').trim().isLength({ min: 3 }),
    (req, res) => {
        const errors = validationResult(req);
        if (!errors.isEmpty()) return res.status(400).json({ errors: errors.array() });
        const tarea = { id: tareas.length + 1, userId: req.user.id, ...req.body, completada: false };
        tareas.push(tarea);
        res.status(201).json(tarea);
    }
);

app.put('/api/tareas/:id', auth, (req, res) => {
    const tarea = tareas.find(t => t.id === parseInt(req.params.id) && t.userId === req.user.id);
    if (!tarea) return res.status(404).json({ error: 'No encontrada' });
    Object.assign(tarea, req.body);
    res.json(tarea);
});

app.delete('/api/tareas/:id', auth, (req, res) => {
    const idx = tareas.findIndex(t => t.id === parseInt(req.params.id) && t.userId === req.user.id);
    if (idx === -1) return res.status(404).json({ error: 'No encontrada' });
    tareas.splice(idx, 1);
    res.status(204).send();
});

app.listen(3000);
```

## Resumen

- **REST**: recursos, verbos HTTP, stateless, URIs únicas
- **JWT**: autenticación stateless con token firmado
- **Middleware**: funciones intermedias (logging, auth, validación)
- **Validación**: librerías como `express-validator` para datos de entrada
- **Documentación**: Swagger/OpenAPI genera UI interactiva de la API
